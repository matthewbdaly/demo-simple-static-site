Emacs Lisp example

;; Markdown
(require 'markdown-mode)
(setq markdown-command "/usr/local/bin/markdown")
(add-to-list 'auto-mode-alist '("\\.markdown$" . markdown-mode))
(add-to-list 'auto-mode-alist '("\\.md$" . markdown-mode))
(setq markdown-css-path "style.css")
